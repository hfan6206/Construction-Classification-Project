## Double data cleaning
def get_all_words(file):
    df = pd.read_excel(file)
    Original = list(df["Processed"])
    ww=[]
    for line in Original:
        words=[]
        line = str(line).lower()
        words_to_add = line.split()
        for w in words_to_add:
            ## Get rid of "/", ".", "%"
            striped_w = w.strip("/").strip(".").strip("-").strip("%")
            words.append(striped_w)
        #print(words)
        _ = " ".join(words)
        ww.append(_)
    df["double_processed"] = pd.DataFrame(ww)
    writer = pd.ExcelWriter(file, engine='xlsxwriter')
    df.to_excel(writer, sheet_name='xxx', index=False)
    writer.save()
    return 1

### function to create 10 excel files to estimate the overall accuracy
def ten_fold(l_original, l_processed):
    length = len(l_original)
    excel_original_list = []
    excel_processed_list = []
    length_of_each = length//10
    for i in range(1,6):
        excel_orinigal = l_original[length_of_each * i : length_of_each * i + 20]
        excel_processed = l_processed[length_of_each * i : length_of_each * i + 20]
        excel_original_list.append(excel_orinigal)
        excel_processed_list.append(excel_processed)
    return excel_original_list, excel_processed_list
